MAX_CASES=20 nohup bash -c '
set -e
for c in alpaca_eval llama_questions reasoning_qa trivia_qa web_questions; do
  echo "[$(date "+%F %T")] running OpenAudioBench $c"
  uv run python test/audio_benchmarks/run_openaudiobench.py \
    --protocol webrtc \
    --component "$c" \
    --max-cases "$MAX_CASES" \
    --early-response-policy warn \
    > "test/audio_benchmarks/openaudiobench_${c}_${MAX_CASES}.log" 2>&1
done
' > test/audio_benchmarks/openaudiobench_20_each.out 2>&1 &

tail -f test/audio_benchmarks/openaudiobench_20_each.out

MAX_CASES=20 nohup bash test/audio_benchmarks/run_voicebench_each.sh \
  > test/audio_benchmarks/voicebench_20_each.out 2>&1 &

tail -f test/audio_benchmarks/voicebench_20_each.out

ps -ef | grep audio_benchmarks | grep -v grep

pkill -f run_openaudiobench.py
pkill -f run_voicebench.py
pkill -f run_voicebench_each.sh
